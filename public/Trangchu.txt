    const elems = {}

    // Utils
    function load(){
      try{
        const raw = localStorage.getItem(STORAGE_KEY)
        state = raw ? JSON.parse(raw) : {topics: []}
      }catch(e){state={topics:[]}}
    }
    function save(){localStorage.setItem(STORAGE_KEY, JSON.stringify(state))}

    function ensureTopic(name){
      let t = state.topics.find(x=>x.topic===name)
      if(!t){t={topic:name,cards:[]};state.topics.push(t);save()}
      return t
    }

    // UI render
    function renderTopics(active){
      const container = elems.topicsList
      container.innerHTML=''
      const all = document.createElement('div')
      all.className='topic '+(active==='__all__'?'active':'')
      all.textContent='Tất cả'
      all.onclick=()=>selectTopic('__all__')
      container.appendChild(all)
      state.topics.forEach(t=>{
        const el=document.createElement('div')
        el.className='topic '+(t.topic===active?'active':'')
        el.textContent=t.topic+' ('+t.cards.length+')'
        el.onclick=()=>selectTopic(t.topic)
        container.appendChild(el)
      })
    }

    function flattenCards(topic){
      if(topic==='__all__') return state.topics.flatMap(t=>t.cards.map(c=>({...c,topic:t.topic})))
      const t = state.topics.find(x=>x.topic===topic)
      return t ? t.cards.map(c=>({...c,topic:topic})) : []
    }

    function renderGrid(topic='__all__'){
      const grid = elems.grid
      const q = document.getElementById('globalSearch').value.trim().toLowerCase()
      const hideKnown = document.getElementById('filterKnown').checked
      let cards = flattenCards(topic)
      if(hideKnown) cards = cards.filter(c=>!c.known)
      if(q) cards = cards.filter(c=> (c.front||'').toLowerCase().includes(q) || (c.back||'').toLowerCase().includes(q))
      grid.innerHTML=''
      elems.currentTopicTitle.textContent = topic==='__all__' ? 'Tất cả flashcards' : topic
      elems.countInfo.textContent = cards.length + ' cards'
      cards.forEach(c=>{
        const wrap=document.createElement('div');wrap.className='card panel'
        const inner=document.createElement('div');inner.className='inner'
        const front=document.createElement('div');front.className='face front'
        const back=document.createElement('div');back.className='face back'
        front.innerHTML=`<div style="font-weight:700;font-size:16px">${escapeHtml(c.front||'')}</div><div class="meta"><div class="small">${escapeHtml(c.topic||'')}</div><div class="small">${c.known? 'Known':'New'}</div></div>`
        back.innerHTML=`<div style="font-size:14px">${escapeHtml(c.back||'')}</div><div style="display:flex;justify-content:space-between;margin-top:12px"><div class="small">ID: ${c.id||''}</div><div><button class="btn ghost" data-action="toggleKnown">${c.known? 'Đánh dấu chưa biết':'Đánh dấu đã biết'}</button> <button class="btn ghost" data-action="edit">Sửa</button> <button class="btn" data-action="delete">Xóa</button></div></div>`
        inner.appendChild(front);inner.appendChild(back);wrap.appendChild(inner)
        inner.onclick = (ev)=>{
          if(ev.target.tagName.toLowerCase()==='button') return // buttons handled separately
          inner.classList.toggle('flipped')
        }
        // button handlers
        back.querySelectorAll('button').forEach(b=>{
          b.addEventListener('click',(ev)=>{
            ev.stopPropagation();
            const act = b.getAttribute('data-action')
            if(act==='toggleKnown'){
              toggleKnown(c)
            }else if(act==='edit'){
              openModalFor(c)
            }else if(act==='delete'){
              deleteCard(c)
            }
          })
        })
        grid.appendChild(wrap)
      })
    }

    function toggleKnown(card){
      const t = state.topics.find(x=>x.topic===card.topic)
      if(!t) return
      const target = t.cards.find(x=>x.id===card.id)
      if(!target) return
      target.known = !target.known
      save();renderTopics(currentTopic);renderGrid(currentTopic)
    }

    function deleteCard(card){
      if(!confirm('Xóa flashcard này?')) return
      const t = state.topics.find(x=>x.topic===card.topic)
      t.cards = t.cards.filter(x=>x.id!==card.id)
      save();renderTopics(currentTopic);renderGrid(currentTopic)
    }

    // modal form
    let currentTopic='__all__'
    function openModalFor(card){
      document.getElementById('modal').style.display='flex'
      document.getElementById('modalTitle').textContent = card ? 'Sửa flashcard' : 'Thêm flashcard'
      document.getElementById('cardTopic').value = card ? card.topic : ''
      document.getElementById('cardFront').value = card ? card.front : ''
      document.getElementById('cardBack').value = card ? card.back : ''
      document.cardToEdit = card || null
    }

    function closeModal(){document.getElementById('modal').style.display='none';document.cardToEdit=null}

    function saveForm(ev){
      ev.preventDefault()
      const topic = document.getElementById('cardTopic').value.trim() || 'General'
      const front = document.getElementById('cardFront').value.trim()
      const back = document.getElementById('cardBack').value.trim()
      if(!front){alert('Mặt trước bắt buộc');return}
      const t = ensureTopic(topic)
      if(document.cardToEdit){
        // update
        const c = t.cards.find(x=>x.id===document.cardToEdit.id)
        c.front=front;c.back=back;save()
      }else{
        const id = 'id_' + Math.random().toString(36).slice(2,9)
        t.cards.unshift({id,front,back,known:false})
        save()
      }
      closeModal();renderTopics(currentTopic);renderGrid(currentTopic)
    }

    function selectTopic(name){currentTopic=name;renderTopics(name);renderGrid(name)}

    // Import JSON
    function importJson(){
      const txt = document.getElementById('uploadArea').value.trim()
      if(!txt) {alert('Dán JSON vào ô upload');return}
      try{
        const data = JSON.parse(txt)
        if(!Array.isArray(data)) throw new Error('Expected array of topics')
        data.forEach(t=>{
          const topicName = t.topic || 'General'
          const target = ensureTopic(topicName)
          if(Array.isArray(t.cards)){
            t.cards.forEach(c=>{
              const id = c.id || ('id_'+Math.random().toString(36).slice(2,9))
              if(!target.cards.some(x=>x.id===id)) target.cards.push({id,front:c.front||'',back:c.back||'',known:!!c.known})
            })
          }
        })
        save();renderTopics(currentTopic);renderGrid(currentTopic);alert('Import thành công')
      }catch(e){alert('Lỗi JSON: '+e.message)}
    }

    function exportJson(){
      const data = state.topics
      const blob = new Blob([JSON.stringify(data,null,2)],{type:'application/json'})
      const url = URL.createObjectURL(blob)
      const a = document.createElement('a');a.href=url;a.download='ai-ielts-flashcards.json';a.click();URL.revokeObjectURL(url)
    }

    function clearStorage(){if(confirm('Xóa toàn bộ dữ liệu local?')){localStorage.removeItem(STORAGE_KEY);state={topics:[]};renderTopics('__all__');renderGrid('__all__')}}

    function escapeHtml(s){return (s||'').replaceAll('&','&amp;').replaceAll('<','&lt;').replaceAll('>','&gt;')}

    // sample data
    function loadSample(){
      const sample = [{topic:'Environment',cards:[{id:'env1',front:'pollution',back:'ô nhiễm'},{id:'env2',front:'sustainable development',back:'phát triển bền vững'}]},{topic:'Education',cards:[{id:'edu1',front:'curriculum',back:'chương trình giảng dạy'}]}]
      state.topics = sample
      save();renderTopics('__all__');renderGrid('__all__')
    }

    // init

      document.getElementById('newTopicBtn').onclick=()=>{
        const name = prompt('Tên chủ đề mới:')

      }

      document.getElementById('closeModal').onclick=closeModal
      document.getElementById('cardForm').addEventListener('submit',saveForm)
      document.getElementById('downloadBtn').onclick=exportJson
      document.getElementById('globalSearch').addEventListener('input',()=>renderGrid(currentTopic))
